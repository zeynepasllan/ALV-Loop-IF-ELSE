*&---------------------------------------------------------------------*
*& Report ZZAS_P04
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
report zzas_p04.

tables: zzas_t02.

types: begin of gty_ogr,
         ogr_no      type zzas_de01,
         ogr_ad      type vorna,
         ogr_soyad   type nachn,
         ders_adi    type vorna,
         birinci_snv type zzas_snv,
         ikinci_snv  type zzas_snv2,
         ort         type char10,
         durum       type char10,
*  S_PNO TYPE CHAR10,
       end of gty_ogr.

data: gt_ogr type table of gty_ogr,
      gs_ogr type gty_ogr.

data: gt_fieldcatalog type slis_t_fieldcat_alv,
      gs_fieldcatalog type slis_fieldcat_alv.

data: gs_layout type slis_layout_alv.

selection-screen begin of block mainsel with frame title text-001.
  select-options: s_pno for zzas_t02-ogr_no.
selection-screen end of block mainsel.

start-of-selection.

  perform : get_data ,
            set_data ,
            show_alv .

end-of-selection .


*&---------------------------------------------------------------------*
*& Form set_data
*&---------------------------------------------------------------------*
form set_data .
*-
  loop at gt_ogr into gs_ogr.
    gs_ogr-ort = ( gs_ogr-birinci_snv + gs_ogr-ikinci_snv ) / 2.
    if gs_ogr-ort >= 45.
      gs_ogr-durum = 'Geçti'.
    else.
      gs_ogr-durum = 'Kaldý'.
    endif.
    modify gt_ogr from gs_ogr.
  endloop.

endform.

*&---------------------------------------------------------------------*
*& Form get_data
*&---------------------------------------------------------------------*
form get_data .
*-
  select * from zzas_t02
    inner join zzas_t03 on zzas_t03~ogr_no eq zzas_t02~ogr_no
    into corresponding fields of table gt_ogr
    where zzas_t02~ogr_no in s_pno.

endform.
*&---------------------------------------------------------------------*
*& Form show_alv
*&---------------------------------------------------------------------*,
form show_alv .
*-
  perform  append_fcat : using 'OGR_NO' 'Öðrenci No' ,
                         using 'OGR_AD' 'Adý' ,
                         using 'OGR_SOYAD' 'Soyadý' ,
                         using 'DERS_ADI' 'Ders Adý' ,
                         using 'BIRINCI_SNV' 'Birinci Sýnav' ,
                         using 'IKINCI_SNV' 'Ýkinci Sýnav' ,
                         using 'ORT' 'Ortalama' ,
                         using 'DURUM' 'Durum' .

  gs_layout-window_titlebar = 'Öðrenci Sýnav Bilgileri'.
  gs_layout-zebra = abap_true.
*gs_layout-colwidth_optimize = abap_true.

  call function 'REUSE_ALV_GRID_DISPLAY'
    exporting
      is_layout   = gs_layout
      it_fieldcat = gt_fieldcatalog
    tables
      t_outtab    = gt_ogr.

endform.

*&---------------------------------------------------------------------*
*& Form append_fcat
*&---------------------------------------------------------------------*
form append_fcat using p_fname p_header .
*-
  clear: gs_fieldcatalog.
  gs_fieldcatalog-fieldname = p_fname.
  gs_fieldcatalog-seltext_s = p_header.
  gs_fieldcatalog-seltext_m = p_header.
  gs_fieldcatalog-seltext_l = p_header.
  append gs_fieldcatalog to gt_fieldcatalog.

endform.
